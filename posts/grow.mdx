var foo = function (bar) {
  return bar++;
};
// A function to add two numbers
function add(a, b) {
  return a + b;
}

// A function to subtract two numbers
function subtract(a, b) {
  return a - b;
}

// A function to multiply two numbers
function multiply(a, b) {
  return a * b;
}

// A function to divide two numbers
function divide(a, b) {
  return a / b;
}

// Calculate the area of a rectangle
function area(width, height) {
  return width * height;
}
var a = 5;
var b = 6;
var c = add(a, b);
var d = subtract(a, b);
var e = multiply(a, b);
var f = divide(a, b);
var g = area(a, b);
console.log(`a + b = ${c}`);
console.log(`a - b = ${d}`);
console.log(`a * b = ${e}`);
console.log(`a / b = ${f}`);
console.log(`a * b = ${g}`);
